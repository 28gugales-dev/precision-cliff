import math
import random

def solve():
    random.seed(42)
    n = 26

    def hexagonal_packing(n_circles, aspect_ratio=1.0):
        """Generate initial hexagonal packing with tunable aspect ratio."""
        positions = []
        sqrt_n = math.sqrt(n_circles)

        # Estimate grid dimensions for hexagonal packing
        rows = int(math.ceil(sqrt_n / math.sqrt(aspect_ratio)))
        cols = int(math.ceil(sqrt_n * math.sqrt(aspect_ratio)))

        row_height = 1.0 / (rows + 0.5) if rows > 1 else 1.0

        for row in range(rows):
            # Hexagonal offset: alternate rows are offset horizontally
            x_offset = 0.25 if row % 2 == 1 else 0.0
            y = (row + 0.5) * row_height

            for col in range(cols):
                if len(positions) >= n_circles:
                    break

                x = (col + 0.5 + x_offset) / (cols + 0.5)
                positions.append([x, y])

            if len(positions) >= n_circles:
                break

        return positions[:n_circles]

    def square_packing(n_circles):
        """Generate initial square packing."""
        positions = []
        side = int(math.ceil(math.sqrt(n_circles)))

        for i in range(n_circles):
            row = i // side
            col = i % side
            x = (col + 0.5) / side
            y = (row + 0.5) / side
            positions.append([x, y])

        return positions

    def optimize_configuration(positions, max_iterations=800):
        """Grow radii iteratively, tracking convergence carefully."""
        positions = [list(p) for p in positions]  # Copy
        radii = [0.001] * n

        for iteration in range(max_iterations):
            max_change = 0.0

            for i in range(n):
                x, y = positions[i]

                # Maximum radius limited by square boundaries
                max_r = min(x, y, 1.0 - x, 1.0 - y)

                # Maximum radius limited by other circles (with small gap for numerical stability)
                for j in range(n):
                    if i != j:
                        xj, yj = positions[j]
                        rj = radii[j]
                        dx = x - xj
                        dy = y - yj
                        dist = math.sqrt(dx * dx + dy * dy)
                        max_r = min(max_r, max(dist - rj - 1e-12, 0.0))

                # Only grow, never shrink
                if max_r > radii[i]:
                    change = max_r - radii[i]
                    max_change = max(max_change, change)
                    radii[i] = max_r

            # Early stopping with tight convergence threshold
            if max_change < 1e-11:
                break

        # Final validation and clamp
        for i in range(n):
            x, y = positions[i]
            radii[i] = min(radii[i], x, y, 1.0 - x, 1.0 - y)

        return positions, radii, sum(radii)

    def perturb_positions(positions, perturbation=0.02):
        """Lightly perturb positions for better packing."""
        perturbed = []
        for x, y in positions:
            dx = random.uniform(-perturbation, perturbation)
            dy = random.uniform(-perturbation, perturbation)
            px = max(0.01, min(0.99, x + dx))

            perturbed.append([px, py])
        return perturbed
